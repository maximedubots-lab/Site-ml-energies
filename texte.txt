    <script>
        // Initialiser les icônes
        lucide.createIcons();

        // Mettre l'année automatiquement
        document.getElementById('year').textContent = new Date().getFullYear();

        // Fonction pour le menu mobile
        function toggleMenu() {
            const menu = document.getElementById('mobile-menu');
            const icon = document.getElementById('menu-icon');
            
            if (menu.classList.contains('hidden')) {
                menu.classList.remove('hidden');
                // Changer l'icône en croix (simulation simple ou re-render lucide si besoin, ici simple toggle visibility)
            } else {
                menu.classList.add('hidden');
            }
        }

        // Fonction pour scroller
        function scrollToSection(id) {
            const element = document.getElementById(id);
            const menu = document.getElementById('mobile-menu');
            
            if (element) {
                element.scrollIntoView({ behavior: 'smooth' });
                // Fermer le menu si ouvert
                if (!menu.classList.contains('hidden')) {
                    menu.classList.add('hidden');
                }
            }
        }

        // Fonction formulaire
        function handleSubmit(e) {
            e.preventDefault();
            const nom = document.getElementById('nom').value;
            const email = document.getElementById('email').value;
            const message = document.getElementById('message').value;

            const subject = `Nouveau message de ${nom} via le site web`;
            const body = `Nom: ${nom}\nEmail: ${email}\n\nMessage:\n${message}`;
            
            window.location.href = `mailto:mlenergiesetetoiles@gmail.com?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
        }
    </script>
